CREATE DATABASE agenda;

USE agenda;

CREATE TABLE familia(
	id int PRIMARY key AUTO_INCREMENT NOT NULL,
    parentesco varchar(200)
);

CREATE TABLE adicionar(
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nome varchar(200),
    tel varchar(11),
    id_parentesco INT,
    
    FOREIGN key (id_parentesco) REFERENCES familia(id)
);
